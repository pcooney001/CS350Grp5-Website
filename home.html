}

class DocumentParser { classifyExternalPages classifyInteralPages

}

class WebsiteWalker { getDirectoryPathFromURL discardDuplicateLinks

}

class ExcelWriter { mehtods to write to excel }

class JSONWriter { mehtods to write on JSON File }

class Script extends Resource{ getUtilizedScriptCount }
