package edu.odu.cs.cs350;

import java.io.File;
import java.nio.file.Path;
import java.nio.file.Files;
import java.util.List;
import java.util.ArrayList;
import java.util.stream.Stream;
import java.util.stream.Collectors;
import java.io.IOException;
import java.nio.file.Path;
import java.nio.file.Paths;


/**
 * Gathers the files necessary before sending them through the Document Parser. 
 */
public class WebsiteWalker 
{   

    private Path offlineDirectoryPath;
    private String websiteURL;

    List<Path> allFiles;
    List<Path> allDirectories; 

    /**
    * Default constructor for the WebsiteWalker
    * @param offlineDirectoryPath
    */
   public WebsiteWalker(Path offlineDirectoryFilePath)
    {
        this.offlineDirectoryPath = offlineDirectoryFilePath;
        this.websiteURL = "";

        this.allFiles = new ArrayList<>();
        this.allDirectories = new ArrayList<>();
    }

     /**
      * Examines the directory to identify all files and subdirectories
      * @throws IOException if directory could not be read 
      */
    public void examineOfflineWebsiteDirectory() throws IOException
    {
       Files.walk(this.offlineDirectoryPath).forEach((Path path) -> {
            if(Files.isRegularFile(path)){
                this.allFiles.add(path);
            }
            else if (Files.isDirectory(path)){
                this.allDirectories.add(path);
            }
        });
    }

    /** 
     * Retrive the offline directory path 
     * @return offlineDirectoryPath
     */
    public Path getOfflineDirectoryPath()
    {
        return offlineDirectoryPath;
    }

    /**
     * Retrive the website URL
     * @return websiteURL
     */
    public String getWebsiteURLs()
    {
        return websiteURL;
    }

     /**
     * Retrive a list of the identified files
     * @return allFiles 
     */
    public List<Path> getFileList()
    {
        return this.allFiles;
    }

    /**
     * Retrive a list of the identified directories
     * @return allDirectories 
     */
    public List<Path> getDirectoryList()
    {
        return this.allDirectories;
    }

}
