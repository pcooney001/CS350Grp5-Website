    public StyleSheet(String newPath)
    {
        super(newPath);

    }

    /**
     * Overload the setClassType to limit Scripts to either Internal or External 
     * Rewrite this with logic to select classification based on provided URLs and Paths 
     * @pre newType A string specifying whether an image is external or internal 
     */
    public void setClassType(String newType)
    {
        switch (newType)
        {
            case "INTERNAL":
                this.classType = Classification.INTERNAL; 
                break; 
            case "EXTERNAL":
                this.classType = Classification.EXTERNAL; 
                break; 
        }

    }
