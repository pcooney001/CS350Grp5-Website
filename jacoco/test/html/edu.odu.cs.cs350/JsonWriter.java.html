        }

    }



    public void CodeDataToFile(ArrayList<String>PathNames, ArrayList<Float> Filesize )
    {

        Iterator<Float> File = Filesize.iterator();
        Iterator<String>Path = PathNames.iterator();

        while (File.hasNext())
        {
            try
            {
            FileWriter texter = new FileWriter(this.filename, true);
            BufferedWriter buffWriter = new BufferedWriter(texter);
            buffWriter.write(File.next() + "C ");
            buffWriter.write(Path.next().toString() +"\n");
            buffWriter.close(); 
            } 

            catch (IOException e) 
            {
                e.printStackTrace();
            }
        }

        float sumofFileSizes =0;
        for(float size: Filesize)

        sumofFileSizes = sumofFileSizes + size;
        

        try 
        {
            FileWriter texter = new FileWriter(this.filename, true);
            BufferedWriter buffWriter = new BufferedWriter(texter);
            buffWriter.write(sumofFileSizes + "C");
            buffWriter.write(".\n");
            buffWriter.close(); 
        } 
        catch (IOException e) 
        {
            e.printStackTrace();
        }
    }
