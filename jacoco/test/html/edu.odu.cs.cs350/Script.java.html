package edu.odu.cs.cs350;

import java.io.IOException;
import java.util.ArrayList;

/**
 * Class to store instances of Scripts found by the Document Parser  
 * 
*/
public class Script extends Resource
{

    /**
     * Default constructor 
     * 
     */
    public Script()
    {
        super(); 
    }

    /**
     * Alternative constructor that is set to a specified Path
     * @param newPath
     */
    public Script(String newPath)
    {
        super(newPath);

    }

    /**
     * Overload the setClassType to limit Scripts to either Internal or External 
     * Rewrite this with logic to select classification based on provided URLs and Paths 
     * 
     * @param newType A string specifying whether an image is external or internal 
     */
    public void setClassType(String newType)
    {
        switch (newType)
        {
            case "INTERNAL":
                this.classType = Classification.INTERNAL; 
                break; 
            case "EXTERNAL":
                this.classType = Classification.EXTERNAL; 
                break; 
        }

    }

}
