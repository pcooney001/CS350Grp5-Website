    /**
     * Default contructor that creates multiple ArrayLists of string type for anchor links, sheets, images, scripts and directory links.
    */
	public HTMLDocument() {
		//first group of ArrayLists for storing Strings
		this.anchorlinks = new ArrayList<String>(); 
		this.stylesheets = new ArrayList<String>();
		this.imagelinks = new ArrayList<String>();
		this.scriptlinks = new ArrayList<String>();
		this.links = new ArrayList<String>();

		//first group of ArrayLists for storing Images, Scripts, Style Sheets, and Resource objects
		this.images = new ArrayList<Image>();
		this.scripts = new ArrayList<Script>();
		this.styleSheets = new ArrayList<StyleSheet>();
		this.archiveFiles = new ArrayList<Resource>();
		this.videoFiles = new ArrayList<Resource>();
		this.audioFiles = new ArrayList<Resource>();
		this.miscFiles = new ArrayList<Resource>();
	}

	/**
	 * Add a anchor tag to HTMLDocument.
	 * 
     * @param newTag the anchor tag in an HTML Document

		this.anchorlinks.add(newTag);
	}
