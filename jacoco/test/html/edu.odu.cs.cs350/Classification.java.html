package edu.odu.cs.cs350;

/**
 * Enumeration class used to classify the local paths of a given 
 * offline, local website.
 */
public enum Classification {
    INTERNAL ("Internal"),
    EXTERNAL("External"),
    INTRA("Intrapage");

    private final String  pathType;

    /**
     * Classification constructor, sets the path type of a local page to the applicable
     * pathType, being either internal, external, or intrapage. 
     * @param Given parameter that sets the classification object's enumerated value.
     */
    Classification(String type){
        this.pathType = type;
    }

    /**
     * Returns the pathType of a given classification enum object.
     */
    public String getPathType(){
        return this.pathType;
    }
}
