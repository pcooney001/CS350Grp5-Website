package edu.odu.cs.cs350;

import java.io.IOException;
import java.math.BigDecimal;
import java.math.RoundingMode;
import java.nio.file.Files;
import java.nio.file.Path;
import java.nio.file.Paths;

public class Image extends Resource 
{
    private double imageSize;

    /**
     * Default constructor 
     */
    public Image()
    {
        super(); 
        this.imageSize = 0.0; 
    }

    /**
     * Alternate constructor that uses a URL provided by the Document Parser 
     * @param newPath
     */
    public Image(String newPath)
    {
        super(newPath);
        this.imageSize = 0.0; 
    }

    /**
     * 
     * Sets the Image Size equal to the size of the file at the specified path in MiB 
     * @param Path
     */
    public void setImageSize(String Path) throws IOException
    {
        Path filePath = Paths.get(Path);
        long fileSizeinBytes = Files.size(filePath);

        double sizeinKB = fileSizeinBytes / 1024; 
        double sizeinMB = sizeinKB / 1024; 
        
        //Output results to 2 decimal places
        BigDecimal bd = new BigDecimal(sizeinMB).setScale(2, RoundingMode.HALF_UP);
        double result = bd.doubleValue(); 
        this.imageSize = result;  
    }

    /**
     * @param newSize 
     */
    public void changeImageSize(double newSize)
    {
        this.imageSize = newSize; 
    }

    /**
     * @return The size of an image in MiB 
     */
    public double getImageSize()
    {
        return this.imageSize; 
    }

    /**
     * Rewrite this with logic to select classification based on provided URLs and Paths 
     * @pre newType A string specifying whether an image is external or internal 
     */
    public void setClassType(String newType)
    {
        switch (newType)
        {
            case "INTERNAL":
                this.classType = Classification.INTERNAL; 
                break; 
            case "EXTERNAL":
                this.classType = Classification.EXTERNAL; 
                break; 
        }

    }


}
