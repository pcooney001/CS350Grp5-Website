package edu.odu.cs.cs350;

import java.io.BufferedWriter;
import java.io.FileWriter;
import java.io.IOException;
import java.text.SimpleDateFormat;
import java.util.ArrayList;
import java.util.Date;
import java.util.Iterator; 

public class TextWriter 
{
    private String filename; 

    /**
     * Default contructor that creates a .txt file with a name based on the time the program was run. 
    */
    public TextWriter()
    {
        Date date = new Date(); 
        SimpleDateFormat ft = new SimpleDateFormat("yyyyMMdd-HHmmss"); 
        this.filename = (ft.format(date) + "-summary.txt");
    }

    /**
     * @param name Creates a .txt document with a prespecified name
     */
    public TextWriter(String name) 
    {
        this.filename = name; 
    }


    /**
     * 
    */
    public void createTextFile()
    {
        try
        {
            FileWriter writer = new FileWriter(this.filename);
            writer.close(); 
        } 
        catch (IOException e) 
        {
            e.printStackTrace();
        } 
   
    }


    /**
     * Update to receive ArrayList of Image Objects for output 
     * @Pre:  Alphabetically sorted PathNames 
     * @Pre:  Corresponding Image Sizes for each webpage converted to MiB 
    */
    public void writeDataToTextFile(ArrayList<String> PathNames, ArrayList<Float> ImageSizes)
    {
        Iterator<Float> itImage = ImageSizes.iterator();
        Iterator<String> itPath = PathNames.iterator(); 
        
        while (itImage.hasNext())
        {
            try 
            {
            FileWriter writer = new FileWriter(this.filename, true);
            BufferedWriter buffWriter = new BufferedWriter(writer);
            buffWriter.write(itImage.next() + "M ");
            buffWriter.write(itPath.next().toString() +"\n");
            buffWriter.close(); 
            } 
            catch (IOException e) 
            {
                e.printStackTrace();
            }
        }

        float SumofImageSizes = 0; 
        for(float size : ImageSizes)
        {
            SumofImageSizes = SumofImageSizes + size; 
        }

        try 
        {
        FileWriter writer = new FileWriter(this.filename, true);
        BufferedWriter buffWriter = new BufferedWriter(writer);
        buffWriter.write(SumofImageSizes + "M ");
        buffWriter.write(".\n");
        buffWriter.close(); 
        } 
        catch (IOException e) 
        {
            e.printStackTrace();
        }

    }

    /**
     * @para name 
    */
    public void setFileName(String name)
    {
        this.filename = name; 
    }

    /**
     * @return A .txt file's name 
    */    
    public String getFileName() 
    {
        return this.filename;
    }




}
